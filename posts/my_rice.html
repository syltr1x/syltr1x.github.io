- Configuración de Hyprland

sudo git clone https://aur.archlinux.org/yay-git.git/ /opt/
sudo chown tuuser:tuuser /opt/yay-git
cd /opt/yay-git
sudo makepkg -si
sudo git clone https://github.com/soldoestech/hyprland.git
sudo chown tuuser:tuuser /opt/hyprland
cd /opt/hyprland
chmod +x set-hypr
./set-hypr
